50void Button::update(const sf::Vector2f mousePos) {
51 //Update the booleans for hover and pressed
52 //Hover
53 this->buttonState = BTN_IDLE;
54 if (this->shape.getGlobalBounds().contains(mousePos)){
55 this->buttonState = BTN_HOVER;
56
57 //Pressed
58 if (sf::Mouse::isButtonPressed(sf::Mouse::Left)){
59 this->buttonState = BTN_ACTIVE;
60 }
61 }
62 switch (this->buttonState){
63 case BTN_IDLE:
64 this->shape.setFillColor(this->idleColor);
65 this->text.setFillColor(this->textIdleColor);
66 break;
67
68 case BTN_HOVER:
69 this->shape.setFillColor(this->hoverColor);
70 this->text.setFillColor(this->textHoverColor);
71 break;
72
73 case BTN_ACTIVE:
74 this->shape.setFillColor(this->activeColor);
75 this->text.setFillColor(this->textActiveColor);
76 break;
77
78 default:
79 this->shape.setFillColor(sf::Color::Red);
80 this->text.setFillColor(sf::Color::Blue);

82 }
