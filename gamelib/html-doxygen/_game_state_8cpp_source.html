47void GameState::updateInput(const float &dt) {
48
49 //Update player input

51 this->player->move(-1.f, 0.f, dt);

53 this->player->move(1.f, 0.f, dt);

55 this->player->move(0.f, -1.f, dt);
56 if (sf::Keyboard::isKeyPressed(sf::Keyboard::Key(this->keybinds.at("MOVE_DOWN"))))
57 this->player->move(0.f, 1.f, dt);
58
59 if (sf::Keyboard::isKeyPressed(sf::Keyboard::Key(this->keybinds.at("CLOSE"))))
